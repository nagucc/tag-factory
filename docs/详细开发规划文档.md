# 数据对象标签管理应用（Tag Factory）详细开发规划文档

## 技术栈选择

- **前端框架**：Next.js v16
- **UI框架**：Ant Design v6
- **后端语言**：Node.js
- **数据库**：MySQL (主数据库)
- **认证**：JWT
- **状态管理**：React Context + useState/useReducer
- **API设计**：RESTful API
- **样式方案**：CSS-in-JS (styled-components) + Antd 主题定制
- **构建工具**：Next.js 内置构建工具
- **部署方案**：Docker + Kubernetes (生产环境)

const sequelize = new Sequelize({

### 数据模型表结构

#### 数据对象定义表
```sql
CREATE TABLE tf_data_objects (
  id                BIGINT PRIMARY KEY AUTO_INCREMENT,
  name              VARCHAR(100) NOT NULL COMMENT '数据对象名称',
  description       TEXT COMMENT '描述',
  data_source_id    BIGINT NOT NULL COMMENT '关联数据源ID',
  query_statement   TEXT NOT NULL COMMENT '查询语句（SQL或MongoDB聚合管道）',
  primary_key       VARCHAR(64) NOT NULL COMMENT '源数据主键字段名',
  display_template  VARCHAR(500) DEFAULT '{{id}}' COMMENT '显示模板，如：{{name}}({{gender}})',
  sync_enabled      BOOLEAN DEFAULT FALSE COMMENT '是否启用同步',
  sync_cron         VARCHAR(100) COMMENT 'CRON表达式',
  sync_strategy     VARCHAR(20) DEFAULT 'full' COMMENT '同步策略：full/incremental',
  last_sync_at      DATETIME COMMENT '最后同步时间',
  last_sync_status  VARCHAR(20) COMMENT '同步状态：success/failed',
  sync_count        INT DEFAULT 0 COMMENT '同步记录数',
  status            VARCHAR(20) DEFAULT 'active' COMMENT '状态：active/disabled',
  created_by        BIGINT NOT NULL,
  created_at        DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at        DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  INDEX idx_data_source (data_source_id),
  INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='数据对象定义表';
```

#### 统一数据记录表
```sql
CREATE TABLE tf_data_records (
  id              BIGINT PRIMARY KEY AUTO_INCREMENT,
  data_object_id  BIGINT NOT NULL COMMENT '关联数据对象ID',
  record_id       VARCHAR(64) NOT NULL COMMENT '记录主键值（对应源数据主键）',
  data            JSON NOT NULL COMMENT '完整原始数据',
  _synced_at      DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '同步时间',
  _sync_status    VARCHAR(20) DEFAULT 'active' COMMENT '同步状态：active/removed',
  UNIQUE KEY uk_object_record (data_object_id, record_id),
  INDEX idx_data_object (data_object_id),
  INDEX idx_record_id (record_id),
  INDEX idx_synced_at (_synced_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='统一数据记录表';
```

#### 标签应用桥接表
```sql
CREATE TABLE tf_tag_applications (
  id              BIGINT PRIMARY KEY AUTO_INCREMENT,
  tag_id          BIGINT NOT NULL COMMENT '标签ID',
  data_object_id  BIGINT NOT NULL COMMENT '数据对象ID',
  record_id       VARCHAR(64) COMMENT '记录ID（为空表示对象级标签）',
  applied_by      BIGINT NOT NULL COMMENT '应用人ID',
  applied_at      DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '应用时间',
  source          VARCHAR(20) DEFAULT 'manual' COMMENT '来源：manual/auto/import',
  status          VARCHAR(20) DEFAULT 'active' COMMENT '状态：active/removed',
  removed_by      BIGINT COMMENT '移除人ID',
  removed_at      DATETIME COMMENT '移除时间',
  INDEX idx_tag (tag_id),
  INDEX idx_data_object (data_object_id),
  INDEX idx_record (record_id),
  INDEX idx_status (status),
  UNIQUE KEY uk_tag_object_record (tag_id, data_object_id, record_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='标签应用桥接表';
```
